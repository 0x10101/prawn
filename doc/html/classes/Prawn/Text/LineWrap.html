      def wrap_line(line, options)
        @document = options[:document]
        @kerning = options[:kerning]
        @width = options[:width]
        @accumulated_width = 0
        @output = ""
        @scan_pattern = @document.font.unicode? ? /\S+|\s+/ : /\S+|\s+/n
        @space_scan_pattern = @document.font.unicode? ? /\s/ : /\s/n

        _wrap_line(line)

        @space_count = @output.count(" ")
        @output
      end
